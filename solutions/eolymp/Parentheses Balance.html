Parentheses Balance

Statement:

( The original statement can be found here )You are given a string consisting of parentheses ( ) and [ ] . A string of this type is said to be correct: Write a program that takes a sequence of strings of this type and check their correctness. Your program can assume that the maximum string length is 128 .

Input:

The first line contains the number of test cases n. Each of the next n lines contains the string of parentheses ( ) and [ ].

Output:

You are given a string consisting of parentheses ( ) and [ ]. A string of this type is said to be correct:

Constraints:

Time limit 1 second
Memory limit 128 MiB
Show problem classificationalgorithms-on-strings

Solution:

#include <iostream>
#include <stack>
#include <string>

using namespace std;

bool isCorrect(const string& parentheses) {
    stack<char> stk;
    
    for (char c : parentheses) {
        if (c == '(' || c == '[') {
            stk.push(c);
        } else if (c == ')' || c == ']') {
            if (stk.empty()) {
                return false;
            }
            
            char top = stk.top();
            stk.pop();
            
            if ((c == ')' && top != '(') || (c == ']' && top != '[')) {
                return false;
            }
        }
    }
    
    return stk.empty();
}

int main() {
    int n;
    cin >> n;
    cin.ignore(); 
    for (int i = 0; i < n; i++) {
        string parentheses;
        getline(cin, parentheses);
        
        if (isCorrect(parentheses)) {
            cout << "Yes" << endl;
        } else {
            cout << "No" << endl;
        }
    }
    
    return 0;
}
