Solution:

a= list(map(int,input().split()))
res=0
for i in range(10):
    for j in range(i+1, 10):
        if(a[i] or a[j]): res+=1
for i in range(10):
    for j in range(i+1, 10):
        for k in range(j+1, 10):
            if(a[i] or a[j] or a[k]): res+=1

print(res%2)
