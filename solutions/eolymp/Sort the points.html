Sort the points

Statement:

( The original statement can be found here )The coordinates of n points are given on a plane. Print them in increasing order of sum of coordinates. In the case of equal sum of point coordinates sort the points in increasing order of abscissa.

Input:

Each line contains the pair of numbers x, y~(0 \le x, y \le 10^9) — the coordinates of one point. The number of dots does not exceed 10^5.

Output:

The coordinates of n points are given on a plane. Print them in increasing order of sum of coordinates. In the case of equal sum of point coordinates sort the points in increasing order of abscissa.

Constraints:

Time limit 3 seconds
Memory limit 128 MiB

Solution:

#include <bits/stdc++.h>
using namespace std;

struct p {
    int x, y;
    int sum;
};

bool cmp(p a, p b) {
    if (a.sum == b.sum) {
        return a.x < b.x;
    }
    return a.sum < b.sum;
}

int main() {
    vector<p> points;

    int x, y;
    while (cin >> x >> y) {
        points.push_back({x, y, x + y});
    }

    sort(points.begin(), points.end(), cmp);

    for (p point : points) {
        cout << point.x << " " << point.y << endl;
    }

    return 0;
}
