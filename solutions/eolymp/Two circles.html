Two circles

Statement:

( The original statement can be found here )Determine at how many points two circles intersect.

Input:

6 numbers x_1, y_1, r_1, x_2, y_2, r_2, where x_1, y_1, x_2, y_2 are the coordinates of the centers of the circles, r_1 , r_2 are their radii.All numbers are real, do not exceed 1000000000 modulo, and are specified with no more than 3 decimal places.

Output:

Determine at how many points two circles intersect.

Constraints:

Time limit 1 second
Memory limit 64 MiB
Show problem classificationgeometry, if-then

Solution:

#include <iostream>
#include <cmath>

using namespace std;

int main() {
    double x1, y1, r1, x2, y2, r2;
    cin >> x1 >> y1 >> r1 >> x2 >> y2 >> r2;

    double d = sqrt(pow(x2 - x1, 2) + pow(y2 - y1, 2));

    if (d > r1 + r2) {
        cout << "0\n"; // the circles do not intersect
    } else if (d < abs(r1 - r2)) {
        cout << "0\n"; // one circle is completely inside the other
    } else if (d == 0 && r1 == r2) {
        cout << "-1\n"; // the circles coincide
    } else {
        int num_common_points = 2; // the circles intersect in two points
        if (d == r1 + r2 || d == abs(r1 - r2)) {
            num_common_points = 1; // the circles intersect in one point
        } else if (d < r1 + r2 && d > abs(r1 - r2)) {
            num_common_points = 2; // the circles intersect in two points
        }
        cout << num_common_points << endl;
    }

    return 0;
}
